527  void getBootstrap(BootstrapAssetGroups &groups, BootstrapAssetConfiguration &config)
528  {
529  uint32_t actual;
530  auto vec = metadata<_SYSMetadataBootAsset>(_SYS_METADATA_BOOT_ASSET, &actual);
531  if (!vec)
532  return;
533 
534  unsigned count = actual / sizeof *vec;
535  for (unsigned i = 0; i != count; ++i) {
536 
537  AssetGroup &group = groups.append();
